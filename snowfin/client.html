191    def _handle_deferred_routine(self, routine: asyncio.Task, request, after: Optional[Callable]):
192        """
193        Create a wrapper for the task supplied and wait on it.
194        log any errors and pass the result onward
195        """
196        async def wrapper():
197            try:
198                response = await routine
199                await self._handle_deferred_response(request, response)
200
201                # call the after callback if it exists
202                if after:
203                    await self._handle_followup_response(request, after)
204            except Exception as e:
205                logger.error(e.__repr__())
206        asyncio.create_task(wrapper())

218    async def _handle_followup_response(self, request, after):
219        """

221        """

235    def infer_response(self, resp: Any) -> _DiscordResponse:
236        kwargs = {}
237        chosen_type = None
238
239        if not isinstance(resp, tuple):
240            resp = [resp]
241            
242        for arg in resp:
243            if isinstance(arg, ResponseType):
244                kwargs['type'] = arg
245
246            elif isinstance(arg, Embed):
247                kwargs.setdefault('embeds', []).append(arg)
248                chosen_type = chosen_type or MessageResponse
249
250            elif is_component(arg):
251                kwargs.setdefault('components', Components()).add_component(arg)
252                chosen_type = chosen_type or (ModalResponse if isinstance(arg, TextInput) else MessageResponse)
253
254            elif isinstance(arg, Components):
255                kwargs['components'] = arg
256
257                for row in arg.rows:
258                    for component in row.components:
259                        if isinstance(component, TextInput):
260                            chosen_type = chosen_type or ModalResponse
261                            break
262                else:
263                    chosen_type = chosen_type or MessageResponse
264
265            elif isinstance(arg, str):
266                kwargs['content'] = arg
267                chosen_type = chosen_type or MessageResponse
268
269            elif isinstance(arg, list):
270                kwargs.setdefault('choices', []).extend(arg)
271                chosen_type = chosen_type or AutocompleteResponse
272
273            elif isinstance(arg, dict):
274                kwargs.update(arg)
275            else:
276                raise ValueError(f"Invalid response type {arg}")
277
278        return (chosen_type or MessageResponse)(**kwargs)

280    async def _handle_request(self, request: Request) -> HTTPResponse:
281        """
282        Grab the callback Coroutine and create a task.
283        """
284        # handle the before requests
285        self.dispatch('before_request', request.ctx)
286
287        func: Optional[Callable] = None
288        after: Optional[Callable] = None
289
290        if request.ctx.type is RequestType.APPLICATION_COMMAND:
291            self.dispatch('command', request.ctx)
292
293            if cmd := self.get_command(request.ctx.data.name):
294                kwargs = {}
295
296                for option in request.ctx.data.options:
297
298                    converted = option.value
299                    if option.type in (OptionType.CHANNEL, OptionType.USER, OptionType.ROLE, OptionType.MENTIONABLE):
300                        converted = request.ctx.data.resolved.get(option.type, option.value)
301                    else:
302                        converted = type(option.value)
303
304                    kwargs[option.name] = converted
305                        
306
307                func = partial(cmd.callback, request.ctx, **kwargs)
308                if cmd.after_callback:
309                    after = partial(cmd.after_callback, request.ctx, **kwargs)
310
311        elif request.ctx.type is RequestType.APPLICATION_COMMAND_AUTOCOMPLETE:
312            self.dispatch('autocomplete', request.ctx)
313
314            if cmd := self.get_command(request.ctx.data.name):
315                for option in request.ctx.data.options:
316                    if option.focused:
317                        callback = cmd.autocomplete_callbacks.get(option.name)
318                        if callback:
319                            func = partial(callback, request.ctx, option.value)
320                        break
321
322        elif request.ctx.type is RequestType.MESSAGE_COMPONENT:
323            self.dispatch('component', request.ctx)
324
325            func, after = self.package_component_callback(
326                request.ctx.data.custom_id,
327                request.ctx.data.component_type,
328                request.ctx
329            )
330
331        elif request.ctx.type is RequestType.MODAL_SUBMIT:
332            self.dispatch('modal', request.ctx)
333
334            if modal := self.modals.get(request.ctx.data.custom_id):
335                func = partial(modal, request.ctx)
336
337                if modal.after_callback:
338                    after = partial(modal.after_callback, request.ctx)
339
340        if self.app.debug: self.log(f"getting callback for {request.ctx.type}: found", f"{func.func.__name__}{func.args[1:]}" if func else None)
341
342        if func:
343            task = asyncio.create_task(func())
344
345            # auto defer if and only if the decorator and/or client told us too and it *can* be defered
346            if self.auto_defer.enabled and \
347                request.ctx.type in (RequestType.APPLICATION_COMMAND, RequestType.MESSAGE_COMPONENT):
348                # we want to defer automatically and keep the original task going
349                # so we wait for up to the timeout, then construct a DeferredResponse ourselves
350                # then handle_deferred_routine() will do the rest
351                done, pending = await asyncio.wait([task], timeout = self.auto_defer.timeout)
352
353
354                if task in pending:
355                    # task didn't return in time, let it keep going and construct a defer for it
356                    resp = DeferredResponse(task,
357                        ephemeral=self.auto_defer.ephemeral
358                    )
359                else:
360                    # the task returned in time, get the result and use that like normal
361                    resp = task.result()
362            else:
363                resp = await task
364        else:
365            return json({"error": "command not found"}, status=404)
366
367        if request.ctx.responded:
368            raise Exception("Callback already responded")
369
370        if not isinstance(resp, DeferredResponse):
371            request.ctx.responded = True
372
373        if not isinstance(resp, (_DiscordResponse, HTTPResponse)):
374            resp = self.infer_response(resp)
375
376        if isinstance(resp, _DiscordResponse):
377            if isinstance(resp, DeferredResponse):
378                # make sure we are sending the correct interaction response type for the request
379                if request.ctx.type == RequestType.MESSAGE_COMPONENT:
380                    resp.type = ResponseType.COMPONENT_DEFER
381                else:
382                    resp.type = ResponseType.DEFER
383
384                # if someone passed in a callable, construct a task for them to keep syntax as clean as possible
385                if not isinstance(resp.task, asyncio.Task):
386                    resp.task = asyncio.create_task(resp.task(self, request.ctx))
387
388                # start or continue the task and post the response to a webhook
389                self._handle_deferred_routine(resp.task, request, after)
390            else:
391                # launch after callbacks if there is any and the command is not a deferred one
392                if after:
393                    asyncio.create_task(self._handle_followup_response(request, after))
394            
395            # do some logging and return the 'dictified' data
396            data = resp.to_dict()
397            if self.app.debug: self.log(f"RESPONDING {request.ctx.type} `{getattr(request.ctx.data, 'name', None)}`", data)
398            return json(data)
399
400        elif isinstance(resp, HTTPResponse):
401            # someone gave us a sanic response, Assume they know what they are doing
402            return resp

408    def load_module(self, module_name: str):
409        resolved_name = importlib.util.resolve_name(module_name, __spec__.parent)
410
411        if resolved_name in self.modules:
412            raise CogLoadError(f"{module_name} already loaded")
413
414        try:
415            module = importlib.import_module(module_name)
416        except Exception as e:
417            raise CogLoadError(f"{module_name} failed to load: {e}")
418        self.modules[resolved_name] = []
419        
420        # go through every class that inherits from Module
421        for cls in inspect.getmembers(module, inspect.isclass):
422            if issubclass(cls[1], Module) and cls[1].enabled and cls[1] != Module:
423                new_module = cls[1](self)
424                self.modules[resolved_name].append(new_module)
425
426                if hasattr(new_module, 'on_load'):
427                    new_module.on_load()
428                
429    def unload_module(self, module: str):
430        module_name = importlib.util.resolve_name(module, __spec__.parent)
431        
432        if module_name not in self.modules:
433            raise ValueError(f"{module_name} not loaded")
434
435        modules: list[Module] = self.modules.pop(module_name)
436
437        for module in modules:
438            if hasattr(module, "on_unload"):
439                module.on_unload()
440
441            # unload all callbacks
442            for callback in module.callbacks:
443                self.remove_callback(callback)
444
445            del module

469        """
470        Gather all callbacks from loaded modules
471        """
472        callbacks = [obj for _, obj in inspect.getmembers(sys.modules["__main__"]) + inspect.getmembers(self) if isinstance(obj, Interactable)]

533    def package_component_callback(self, custom_id: str, component_type: ComponentType, ctx: Interaction) -> Callable:
534         # loop through all all our registered component callbacks
535        for (_id, _type), callback in self.components.items():
536
537            # check the type first and foremost
538            if _type == component_type:
539
540                kwargs = {}
541
542                # make sure there are actually mappings to check
543                if None not in (callback.mappings, callback.chopped_id):
544                    just_values = []
545
546                    left = custom_id
547
548                    # go through all the constants in the defined custom_id and
549                    # check if they match the mappings. Construct a list of the
550                    # values to pass to the callback and convert
551                    for i in range(len(callback.chopped_id)):
552                        
553                        # this is the next constant in the custom_id
554                        segment = callback.chopped_id[i]
555
556                        # make sure the constant is in the custom_id
557                        if segment not in left:
558                            break
559
560                        # strip the constant from the custom_id so we know that
561                        # the next part of the string is the value
562                        left = left.removeprefix(segment)
563                        if i+1 < len(callback.mappings):
564                            value = left.strip(callback.chopped_id[i+1])[0]
565                        else:
566                            value = left
567                        
568                        just_values.append(value)
569
570                        # remove the value from the custom_id so we know
571                        # that the next part of the string is the next constant
572                        left = left.removeprefix(value)
573                            
574                    # check to make sure that we have the right number of values collected
575                    if len(just_values) != len(callback.mappings):
576                        continue
577
578                    mappings = callback.mappings.items()
579                    for i, (name, _type) in enumerate(mappings):
580                        # convert the value to the correct type if possible
581
582                        kwargs[name] = just_values[i]
583
584                        with suppress(ValueError):
585                            kwargs[name] = _type(kwargs[name])
586                elif _id != custom_id:
587                        continue
588
589
590                return (
591                    functools.partial(callback.callback, ctx, **kwargs),
592                    functools.partial(callback.after_callback, ctx, **kwargs) if callback.after_callback else None
593                )
594
595        return None, None

281    async def _handle_request(self, request: Request) -> HTTPResponse:
282        """
283        Grab the callback Coroutine and create a task.
284        """
285        # handle the before requests
286        self.dispatch('before_request', request.ctx)
287
288        func: Optional[Callable] = None
289        after: Optional[Callable] = None
290
291        if request.ctx.type is RequestType.APPLICATION_COMMAND:
292            self.dispatch('command', request.ctx)
293
294            if cmd := self.get_command(request.ctx.data.name):
295                kwargs = {}
296
297                for option in request.ctx.data.options:
298
299                    converted = option.value
300                    if option.type in (OptionType.CHANNEL, OptionType.USER, OptionType.ROLE, OptionType.MENTIONABLE):
301                        converted = request.ctx.data.resolved.get(option.type, option.value)
302                    else:
303                        converted = type(option.value)
304
305                    kwargs[option.name] = converted
306                        
307
308                func = partial(cmd.callback, request.ctx, **kwargs)
309                if cmd.after_callback:
310                    after = partial(cmd.after_callback, request.ctx, **kwargs)
311
312        elif request.ctx.type is RequestType.APPLICATION_COMMAND_AUTOCOMPLETE:
313            self.dispatch('autocomplete', request.ctx)
314
315            if cmd := self.get_command(request.ctx.data.name):
316                for option in request.ctx.data.options:
317                    if option.focused:
318                        callback = cmd.autocomplete_callbacks.get(option.name)
319                        if callback:
320                            func = partial(callback, request.ctx, option.value)
321                        break
322
323        elif request.ctx.type is RequestType.MESSAGE_COMPONENT:
324            self.dispatch('component', request.ctx)
325
326            func, after = self.package_component_callback(
327                request.ctx.data.custom_id,
328                request.ctx.data.component_type,
329                request.ctx
330            )
331
332        elif request.ctx.type is RequestType.MODAL_SUBMIT:
333            self.dispatch('modal', request.ctx)
334
335            if modal := self.modals.get(request.ctx.data.custom_id):
336                func = partial(modal, request.ctx)
337
338                if modal.after_callback:
339                    after = partial(modal.after_callback, request.ctx)
340
341        if self.app.debug: self.log(f"getting callback for {request.ctx.type}: found", f"{func.func.__name__}{func.args[1:]}" if func else None)
342
343        if func:
344            task = asyncio.create_task(func())
345
346            # auto defer if and only if the decorator and/or client told us too and it *can* be defered
347            if self.auto_defer.enabled and \
348                request.ctx.type in (RequestType.APPLICATION_COMMAND, RequestType.MESSAGE_COMPONENT):
349                # we want to defer automatically and keep the original task going
350                # so we wait for up to the timeout, then construct a DeferredResponse ourselves
351                # then handle_deferred_routine() will do the rest
352                done, pending = await asyncio.wait([task], timeout = self.auto_defer.timeout)
353
354
355                if task in pending:
356                    # task didn't return in time, let it keep going and construct a defer for it
357                    resp = DeferredResponse(task,
358                        ephemeral=self.auto_defer.ephemeral
359                    )
360                else:
361                    # the task returned in time, get the result and use that like normal
362                    resp = task.result()
363            else:
364                resp = await task
365        else:
366            return json({"error": "command not found"}, status=404)
367
368        if request.ctx.responded:
369            raise Exception("Callback already responded")
370
371        if not isinstance(resp, DeferredResponse):
372            request.ctx.responded = True
373
374        if not isinstance(resp, (_DiscordResponse, HTTPResponse)):
375            resp = self.infer_response(resp)
376
377        if isinstance(resp, _DiscordResponse):
378            if isinstance(resp, DeferredResponse):
379                # make sure we are sending the correct interaction response type for the request
380                if request.ctx.type == RequestType.MESSAGE_COMPONENT:
381                    resp.type = ResponseType.COMPONENT_DEFER
382                else:
383                    resp.type = ResponseType.DEFER
384
385                # if someone passed in a callable, construct a task for them to keep syntax as clean as possible
386                if not isinstance(resp.task, asyncio.Task):
387                    resp.task = asyncio.create_task(resp.task(self, request.ctx))
388
389                # start or continue the task and post the response to a webhook
390                self._handle_deferred_routine(resp.task, request, after)
391            else:
392                # launch after callbacks if there is any and the command is not a deferred one
393                if after:
394                    asyncio.create_task(self._handle_followup_response(request, after))
395            
396            # do some logging and return the 'dictified' data
397            data = resp.to_dict()
398            if self.app.debug: self.log(f"RESPONDING {request.ctx.type} `{getattr(request.ctx.data, 'name', None)}`", data)
399            return json(data)
400
401        elif isinstance(resp, HTTPResponse):
402            # someone gave us a sanic response, Assume they know what they are doing
403            return resp

566                        else:

572                        # that the next part of the string is the next constant
573                        left = left.removeprefix(value)
574

576                    if len(just_values) != len(callback.mappings):
